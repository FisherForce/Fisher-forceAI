// === CARTE MIGRATEURS PRIVÉE (100% fonctionnelle dès maintenant) === let migrateursMap = null; let migrateursMarkers = {}; function initMigrateursMap() { if (migrateursMap) migrateursMap.remove(); migrateursMap = L.map('migrateursMap').setView([46.8, -1.0], 7); L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {attribution: '© OpenStreetMap'}).addTo(migrateursMap); // Charge tes spots migrateurs depuis localStorage const myMigrateurs = JSON.parse(localStorage.getItem('myMigrateursSpots') || '[]'); myMigrateurs.forEach(spot => { const key = spot.lat + '_' + spot.lng; if (migrateursMarkers[key]) return; migrateursMarkers[key] = L.marker([spot.lat, spot.lng], { icon: L.divIcon({ html: `
Fish
`, iconSize: [50, 50], iconAnchor: [25, 50] }) }).addTo(migrateursMap) .bindPopup(` Spot Migrateur
${spot.name}
Espèce : ${spot.species}
Ajouté le ${spot.date} `); }); // Double-clic = ajouter un spot migrateur migrateursMap.on('dblclick', e => { const speciesList = ["Saumon", "Alose", "Truite de mer", "Anguille", "Lamproie", "Esturgeon", "Mulet", "Autre"]; const species = prompt("Espèce migratrice ?", speciesList[0]); if (!species) return; const name = prompt("Nom du spot ?", "Spot saumon secret"); if (!name) return; const newSpot = { name: name.trim(), lat: e.latlng.lat.toFixed(6), lng: e.latlng.lng.toFixed(6), species: species, date: new Date().toLocaleDateString('fr-FR') }; const spots = JSON.parse(localStorage.getItem('myMigrateursSpots') || '[]'); spots.push(newSpot); localStorage.setItem('myMigrateursSpots', JSON.stringify(spots)); alert(`Spot migrateur "${name}" ajouté !`); initMigrateursMap(); }); // Géolocalisation if ("geolocation" in navigator) { navigator.geolocation.getCurrentPosition(pos => { migrateursMap.setView([pos.coords.latitude, pos.coords.longitude], 10); L.marker([pos.coords.latitude, pos.coords.longitude], { icon: L.divIcon({html: '
Person
', iconSize: [50,50], iconAnchor: [25,50]}) }).addTo(migrateursMap).bindPopup("Tu es ici").openPopup(); }); } } // Ouvre la carte migrateurs au chargement (ou au clic si tu veux un bouton) document.addEventListener('DOMContentLoaded', initMigrateursMap);
